Maldives - max annual Sea Surface Temperature

Interactive maps of mean marine heatwave intensity (1981-present) from NOAA 1/4° Daily Optimum Interpolation Sea Surface Temperature (OISST).

library(terra)
library(tidyverse)

mhw_intensity <- rast("/Users/rof011/rOceanData/data/maldives/Bleaching-mhw_annual-mean_intensity.tiff")

library(leaflet)

num_colors <- 10  # Adjust the number of colors for granularity
rdbu_palette <- colorNumeric(palette = "Reds", domain = values(mhw_intensity), n = num_colors)

# Leaflet Map with Raster Layer
leaflet() %>%
  addTiles() %>%
  setView(lat = 3.9, lng = 73.5, zoom = 8) %>%
  addProviderTiles('Esri.WorldImagery', group = "Base Map", options=providerTileOptions(maxNativeZoom=18,maxZoom=100)) |>
  addRasterImage(mhw_intensity,group = "SST", colors = rdbu_palette, opacity = 0.8) %>%
  addLegend(pal = rdbu_palette, values = values(mhw_intensity), title = "Marine heatwave intensity", opacity = 0.8) |> 
  addLayersControl(
    baseGroups = c("Base Map"),
    overlayGroups = c("SST"),
    options = layersControlOptions(collapsed = FALSE)
  )
